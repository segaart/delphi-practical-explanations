Parallel Arrays in Delphi

Level 1: Totalling & Average

Unit1.pas
var
arrNumbers: array[1..5] of Integer;
arrNames: array[1..5] of String;
iLoop: Integer;
iTotal: Integer;
rAverage: Real;
 
begin
// Initialize total
iTotal := 0;
 
// Loop through array
for iLoop := 1 to 5 do
begin
// Add current element to total
iTotal := iTotal + arrNumbers[iLoop];
end;
 
// Calculate average
rAverage := iTotal / 5;
 
// Display results
ShowMessage('Total: ' + IntToStr(iTotal));
ShowMessage('Average: ' + FloatToStr(rAverage));
end;

var
arrAges: array[1..5] of Integer;
arrNames: array[1..5] of String;
iLoop: Integer;
iMaxAge: Integer;
sOldestPerson: String;
 
begin
// Initialize maximum age from first element
iMaxAge := arrAges[1];
sOldestPerson := arrNames[1];
 
// Find maximum age
for iLoop := 1 to 5 do
begin
// Compare with current maximum
if arrAges[iLoop] > iMaxAge then
begin
iMaxAge := arrAges[iLoop];
sOldestPerson := arrNames[iLoop];
end;
end;
 
// Display result (outside loop)
ShowMessage(sOldestPerson + ' is oldest: ' +
IntToStr(iMaxAge) + ' years');
end;

var
arrAges: array[1..5] of Integer;
arrNames: array[1..5] of String;
iLoop: Integer;
iCount: Integer;
iThreshold: Integer;
 
begin
// Initialize counter
iCount := 0;
iThreshold := 40;
 
// Count people younger than threshold
for iLoop := 1 to 5 do
begin
// Check if age is less than threshold
if arrAges[iLoop] < iThreshold then
iCount := iCount + 1;
end;
 
// Display result (outside loop)
ShowMessage(IntToStr(iCount) +
' people are younger than ' +
IntToStr(iThreshold));
end;
